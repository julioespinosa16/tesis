% --- Slide 4: Marco Teórico (Corregida: Contraste de Integralidad) ---
\begin{frame}{Marco Teórico: La Pesadilla de la Integralidad}
    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            \begin{block}{1. Relajación Lineal (PL)}
                \small
                \begin{equation*}
                    \begin{aligned}
                        & \min \quad c^\top x \\
                        & \text{s. a.} \quad Ax = b \\
                        & \quad \quad \quad \mathbf{x \ge 0} 
                    \end{aligned}
                \end{equation*}
            \end{block}
            \begin{itemize}
                \item \footnotesize \textbf{Complejidad:} Polinomial (Simplex/Punto Interior).
                \item \footnotesize \textbf{Dominio:} Espacio continuo (Poliedro convexo).
            \end{itemize}
        \end{column}

        \begin{column}{0.48\textwidth}
            \begin{block}{2. Programación Entera (PLE)}
                \small
                \begin{equation*}
                    \begin{aligned}
                        & \min \quad c^\top x \\
                        & \text{s. a.} \quad Ax = b \\
                        & \quad \quad \quad \mathbf{x \in \mathbb{Z}_{+}^n} 
                    \end{aligned}
                \end{equation*}
            \end{block}
            \begin{itemize}
                \item \footnotesize \textbf{Complejidad:} NP-Duro.
                \item \footnotesize \textbf{Reto:} Búsqueda en espacios discretos.
            \end{itemize}
        \end{column}
    \end{columns}

    \vspace{0.4cm}
    \begin{itemize}
        \item El \textbf{Integrality Gap} es la distancia entre el óptimo relajado y el óptimo entero.
        \item En problemas de gran escala ($n > 10,000$), algoritmos como \textit{Branch and Bound} se vuelven computacionalmente prohibitivos.
    \end{itemize}

    \centering
    \vspace{0.6cm}
    {\Large\bfseries\alert{"La discreción es el enemigo de la velocidad... a menos que conozcas la estructura".}}
\end{frame}


% --- Slide 5: El "As" bajo la manga (Versión con Kerning Corregido) ---
\begin{frame}{El ``As'' bajo la manga: Unimodularidad Total (TU)}

    \begin{columns}[T]
        % Columna 1: El fundamento técnico
        \begin{column}{0.55\textwidth}
            \begin{definition}[Unimodularidad Total]
                \small Una matriz $A$ es TU si el determinante de \textbf{toda} submatriz cuadrada pertenece al conjunto $\{-1, 0, 1\}$.
            \end{definition}

            \vspace{0.3cm}

            \begin{theorem}
                \small Si $A$ es totalmente unimodular y $b$ es un vector entero, todos los vértices del poliedro $\{x \mid Ax=b, x \ge 0\}$ son enteros.
            \end{theorem}

            \vspace{0.5cm}
            \textbf{Consecuencia para el proyecto:}
            \begin{itemize}
                \item El algoritmo Simplex resuelve directamente la instancia entera.
                \item Eliminamos el costo exponencial de \textit{Branch \& Bound}.
                \item Viabilidad para decenas de miles de variables.
            \end{itemize}
        \end{column}

        % Columna 2: Visualización y Sentencia
        \begin{column}{0.42\textwidth}
            \centering
            \includegraphics[width=0.9\textwidth]{simplex_plot.png}

            \alert{\Large\bfseries ``Cuando la estructura es perfecta, la relajación lineal es la solución exacta.''}
        \end{column}
    \end{columns}
\end{frame}

% --- Slide 6: Fotomosaicos (Modelo con dimensiones m x n) ---
\begin{frame}{Fotomosaicos: El Modelo de Asignación Global}

    \begin{columns}[T]
        % Columna 1: El modelo con sumatorias explícitas
        \begin{column}{0.55\textwidth}
            Dada una imagen objetivo discretizada en una malla de $m \times n$ celdas y un catálogo $F$:
            
            \begin{block}{Minimización de Distancia Global}
                \small
                \begin{equation*}
                    \min \sum_{f \in F} \sum_{i=1}^{m} \sum_{j=1}^{n} d(b_f, \beta_{ij})^2 \, x_{fij}
                \end{equation*}
            \end{block}
            
            \textbf{Sujeto a:}
            \begin{itemize}
                \item \footnotesize $\sum_{f \in F} x_{fij} = 1, \quad \forall i, j$ \hfill \textit{(Cobertura)}
                \item \footnotesize $\sum_{i,j} x_{fij} \le u_f, \quad \forall f \in F$ \hfill \textit{(Inventario)}
                \item \footnotesize $x_{fij} \in \{0, 1\}$ \hfill \textit{(Asignación)}
            \end{itemize}
        \end{column}

        % Columna 2: Espacios Métricos y Sentencia
        \begin{column}{0.42\textwidth}
            \textbf{Robustez del modelo:}
            \vspace{0.2cm}
            \small
            La métrica $d$ opera en un espacio normado $\mathbb{R}^k$ según la representación:
            \begin{itemize}
                \item \textbf{Grises ($k=1$):} Distancia absoluta en luminancia.
                \item \textbf{Color ($k=3$):} Norma euclidiana en RGB o CIELAB.
                \item \textbf{Impresión ($k=4$):} Espacio CMYK.
            \end{itemize}

            \vspace{1.0cm}

            % Homologado con comillas tipográficas y color naranja Opt-Art
            \alert{\Large\bfseries ``Buscamos el mínimo global, no el vecino más cercano.''}
        \end{column}
    \end{columns}
\end{frame}

% --- Slide 8: La Carpintería Algorítmica (Mapeo eta y lpSolve) ---
\begin{frame}{La Carpintería Algorítmica: Mapeo $\eta$ y Sparsidad}

    \begin{columns}[T]
        % Columna 1: El Algoritmo con la eta griega
        \begin{column}{0.55\textwidth}
            ¿Cómo procesar un tensor 3D de costos ($C_{fij}$) en un solucionador lineal estándar?
            
            \vspace{0.2cm}
            \textbf{Vectorización Row-wise (Mapeo $\eta$):}
            \begin{algorithm}[H]
                \small
                \floatname{algorithm}{Algoritmo}
                \begin{algorithmic}[1]
                    % --- AQUÍ ESTÁ EL CAMBIO DE LA ETA GRIEGA ---
                    \Procedure{Mapeo$\eta$}{$f, i, j, m, n$}
                        \State $t \gets (f - 1) \cdot (mn) + (i-1) \cdot n + j$
                        \State \Return $t$
                    \EndProcedure
                \end{algorithmic}
            \end{algorithm}
            \footnotesize Este mapeo biyectivo transforma el tensor de costos en un vector compatible con la función objetivo $c^\top x$ del modelo.
        \end{column}

        % Columna 2: Implementación en R
        \begin{column}{0.42\textwidth}
            \textbf{Arquitectura en R:}
            \begin{itemize}
                \item \textbf{Solucionador:} Uso de la librería \texttt{lpSolve} para optimización lineal masiva.
                \item \textbf{Estructura Sparse:} La matriz de restricciones resultante es altamente rala.
                \item \textbf{Escalabilidad:} Permite manejar más de \textbf{72,000 variables} con 16 GB de RAM.
            \end{itemize}

            \vspace{1.0cm}

            % Frase homologada en naranja Opt-Art
            \alert{\Large\bfseries ``La eficiencia vive en los ceros de la matriz.''}
        \end{column}
    \end{columns}
\end{frame}

% --- Slide 12: El Motor del TSP (Teoría vs. Heurística) ---
\begin{frame}{El Motor del TSP: Rigor Teórico vs. Eficiencia Heurística}

    \begin{columns}[T]
        % Columna 1: El límite de la teoría (MTZ)
        \begin{column}{0.5\textwidth}
            \textbf{Modelo Teórico (MTZ):}
            \small Eliminación de \textit{subtours} mediante variables de orden $u_i$.
            \begin{equation*}
                u_i - u_j + (\tau-1) x_{ij} \le \tau-2 \quad \forall i \neq j
            \end{equation*}
            
            \begin{itemize}
                \item \textbf{Complejidad:} Genera $O(\tau^2)$ restricciones.
                \item \textbf{Limitación:} Computacionalmente intratable para $\tau > 15,000$ ciudades.
            \end{itemize}
        \end{column}

        % Columna 2: La potencia de la heurística
        \begin{column}{0.48\textwidth}
            \textbf{Implementación Práctica:}
            \begin{enumerate}
                \item \textbf{Inicialización:} Inserción Arbitraria para obtener una solución inicial factible.
                \item \textbf{Refinamiento:} Algoritmo \textbf{2-opt} para optimización local.
            \end{enumerate}
            \vspace{0.4cm}
            \small Este enfoque transforma un problema NP-Duro en un proceso convergente de alta velocidad.
        \end{column}
    \end{columns}

    \vfill % Balanceo vertical

\begin{frame}{El Arte de Desanudar: El Efecto 2-opt}
    \begin{columns}[T]
        % Columna 1: Fundamento Geométrico
        \begin{column}{0.5\textwidth}
            El algoritmo \textbf{2-opt} intercambia dos aristas existentes por dos nuevas siempre que el costo total disminuya.
            
            \vspace{0.4cm}
            \begin{itemize}
                \item Se fundamenta en la \textbf{Desigualdad Triangular} euclidiana: ``descruzar'' aristas siempre acorta la ruta total.
                \item Elimina el ruido visual de las intersecciones, ``peinando'' el trazo de forma iterativa.
                \item \textbf{Costo:} $O(\tau^2)$ por iteración, logrando convergencia rápida hacia óptimos locales de alta calidad.
            \end{itemize}
        \end{column}

\begin{frame}{Anexo A: Algoritmo de Partición Modular}
    \begin{algorithm}[H]
        \floatname{algorithm}{Algoritmo A1}
        \caption{Partición modular simétrica (\texttt{matsplitter})}
        \begin{algorithmic}[1]
            \Procedure{MatSplitter}{MatrizOriginal, m, n}
            \State $\mu \gets \text{nrow(MatrizOriginal)},\ \nu \gets \text{ncol(MatrizOriginal)}$
            \State $\mu' \gets \mu - (\mu \mod m)$ \Comment{Ajuste de filas}
            \State $\nu' \gets \nu - (\nu \mod n)$ \Comment{Ajuste de columnas}
            \State MatrizRecortada $\gets$ MatrizOriginal$[1:\mu', 1:\nu']$
            \State $k \gets \mu'/m,\ h \gets \nu'/n$ \Comment{Dimensiones del tile}
            \State ListaRectángulos $\gets$ Dividir(MatrizRecortada, k, h)
            \State \Return ListaRectángulos
            \EndProcedure
        \end{algorithmic}
    \end{algorithm}
    Garantiza bloques íntegros y evita sesgos de luminosidad en los bordes.
\end{frame}

% --- Slide 24: Anexo B: Extensión a Dominós de Bosch ---
\begin{frame}{Anexo B: Extensión a Dominós de Bosch}
    \textbf{El Reto:} Cubrir la región de la imagen mediante fichas de dominó ($2 \times 1$).
    
    \vspace{0.4cm}
    \begin{itemize}
        \item Se modela como un problema de \textbf{Emparejamiento Perfecto} en una gráfica de rejilla.
        \item Requiere la introducción de variables binarias de orientación: $x_{ij}^H$ (horizontal) y $x_{ij}^V$ (vertical).
    \end{itemize}

    \begin{block}{Restricción de Cobertura}
        Cada celda $(i,j)$ debe ser cubierta por exactamente una mitad de una ficha:
        \begin{equation*}
            x_{ij}^H + x_{i,j-1}^H + x_{ij}^V + x_{i-1,j}^V = 1
        \end{equation*}
    \end{block}

    \vspace{0.3cm}
    \small El mapeo vectorial $\eta$ desarrollado en esta tesis es lo suficientemente robusto para manejar la expansión dimensional de este espacio de variables.

    \vfill

    \vspace{0.4cm}
\end{frame}
