% --- Slide 5: Marco Teórico I: Programación Lineal (LP) ---
\begin{frame}{Marco Teórico I: Programación Lineal (LP)}
    \begin{columns}[T]
        \begin{column}{0.55\textwidth}
            \textbf{Definición General:}
            Busca optimizar una función objetivo sujeta a restricciones de igualdad o desigualdad, todas de carácter lineal.
                 
\begin{block}{Forma Estándar}
                \small
                \begin{equation*}
                    \begin{array}{r@{\quad}llll}
                        \text{min}  & z & = & \displaystyle\sum_{i=1}^{n} c_i x_i & \\[0.4cm]
                        \text{s.a.} &   &   & \displaystyle\sum_{i=1}^{n} a_{ij}x_i \le b_j, \quad \forall j & \equiv \quad \mathbf{A}\mathbf{x} \le \mathbf{b} \\[0.4cm]
                                    &   &   & \mathbf{x} \ge \mathbf{0} & \text{con } \mathbf{A} \in \mathbb{R}^{m \times n}
                    \end{array}
                \end{equation*}
            \end{block}
            
            
            \begin{itemize}
                \item \footnotesize \textbf{Geometría:} La región factible es un poliedro convexo.
                \item \footnotesize \textbf{Resolución:} Algoritmo \textbf{Simplex} (Dantzig, 1947).
                \item \footnotesize \textbf{Complejidad:} Eficiente (Tiempo polinomial).
            \end{itemize}
        \end{column}

        \begin{column}{0.42\textwidth}
            \centering
            \vspace{0.5cm}
            \includegraphics[width=\textwidth]{simplex_plot.png}
            \vspace{0.4cm}
            \scriptsize \textit{El óptimo siempre se encuentra en un vértice del poliedro.}
        \end{column}
    \end{columns}
\end{frame}



% --- Slide 6: Marco Teórico II: El Desafío de la Integralidad ---
\begin{frame}{Marco Teórico II: El Desafío de la Integralidad}
    \begin{columns}[T]
        % Columna Izquierda: Teoría y Retos
        \begin{column}{0.52\textwidth}
            Al exigir $\mathbf{x \in \mathbb{Z}^n}$, el problema pierde su estructura continua:
            
            \vspace{0.3cm}
            \begin{itemize}
                \item \small \textbf{Fallo del Redondeo:} La solución óptima entera puede ser infactible o estar muy alejada del óptimo relajado.
                \item \small \textbf{Explosión Exponencial:} La complejidad es \textbf{NP-Dura}. Resolver una instancia de $n$ variables binarias implica, en teoría, explorar hasta $2^n$ combinaciones.
            \end{itemize}

            \begin{block}{Complejidad del Branch \& Bound}
                \footnotesize 
                Aunque es una exploración inteligente, su estructura es la de un \textbf{árbol de búsqueda}. Este método divide iterativamente el problema original en subproblemas mediante restricciones adicionales (\textit{branching}) y descarta ramas mediante cotas lineales (\textit{bounding}). Ante $n$ grande, el número de ramificaciones y nodos a resolver crece exponencialmente, volviéndose computacionalmente prohibitivo (por memoria y tiempo).
            \end{block}
        \end{column}

        % Columna Derecha: Intuición Visual (TikZ Corregido)
        \begin{column}{0.45\textwidth}
            \centering
            \textbf{\footnotesize La Falacia del Redondeo} \\

% --- Slide 7: El Puente: Unimodularidad Total (TU) ---
\begin{frame}{El Puente: Unimodularidad Total (TU)}
    \begin{columns}[T]
        % Columna Izquierda: Definiciones y Teorema (58%)
        \begin{column}{0.58\textwidth}
            ¿Cómo resolvemos instancias con órdenes de miles de variables enteras? Aprovechando la \textbf{estructura de la matriz $A$}.

            \vspace{0.2cm}

            \begin{block}{Definición (Unimodularidad Total)}
                \small Una matriz $A$ es TU si el determinante de \textbf{toda} submatriz cuadrada pertenece al conjunto $\{-1, 0, 1\}$.
            \end{block}

            \vspace{0.1cm}

            \begin{block}{Teorema (Hoffman y Kruskal)}
                \small Si $A$ es TU y $b$ es un vector entero, todos los vértices del poliedro $\{x \mid Ax=b, x \ge 0\}$ son \textbf{enteros}.
            \end{block}
        \end{column}

        % Columna Derecha: Aplicación y Cierre (38%)
        \begin{column}{0.38\textwidth}
            \textbf{Relevancia en Opt-Art:}
            \begin{itemize}
                \item \footnotesize \textbf{Mosaicos:} La matriz de asignación es TU por construcción.
                \item \footnotesize \textbf{Rutas:} Ciertas redes de flujo garantizan integralidad.
                \item \footnotesize \textbf{Velocidad:} El Simplex resuelve el problema entero directamente.
            \end{itemize}
            
            \vfill 

            \begin{flushleft}
                \alert{\Large\bfseries ``Cuando la estructura es perfecta, la relajación lineal es la solución exacta.''}
            \end{flushleft}
            \vspace{0.4cm}
        \end{column}
    \end{columns}
\end{frame}

% --- Slide 8: Fotomosaicos (Modelo Global y Estructura Modular) ---
\begin{frame}{Fotomosaicos: Modelo Global y Estructura Modular}
    \vspace{0.2cm} % Ajuste inicial para despegar del header

    \begin{columns}[T, onlytextwidth]
        % Columna 1: El Modelo Matemático (52%)
        \begin{column}{0.52\textwidth}
            \textbf{\small Formulación del Modelo:} \\
            \footnotesize Dada una imagen objetivo en una malla de $m \times n$ y un catálogo $F$:
            
            \vspace{0.3cm}
            \begin{block}{Minimización de Distancia Global}
                \small
                \begin{equation*}
                    \begin{array}{ll}
                        \text{min} & \displaystyle\sum_{f \in F} \sum_{i=1}^{m} \sum_{j=1}^{n} d(b_f, \beta_{ij})^2 \, x_{fij}
                    \end{array}
                \end{equation*}

% --- Slide 13: El Motor del TSP (Formulación y Heurística) ---
\begin{frame}{El Motor del TSP: De la Teoría a la Práctica}
    \begin{columns}[T]
        % Columna 1: La Teoría (Alineada a la Tesina)
        \begin{column}{0.52\textwidth}
        \textbf{Modelo Clásico (DFJ\footnote[frame]{Dantzig, Fulkerson y Johnson (1954): Formulación clásica que utiliza las restricciones SEC para garantizar un tour hamiltoniano.}):}
            \footnotesize
            \begin{equation*}
                \begin{array}{ll}
                    \text{min} & \displaystyle\sum_{i,j} c_{ij} x_{ij} \\[0.3cm]
                    \text{s.a.} & \displaystyle\sum_{i} x_{ij} = 1, \quad \sum_{j} x_{ij} = 1
                \end{array}
            \end{equation*}

            
            \textbf{Restricciones de Subtour (SEC):}
            \begin{equation*}
                \sum_{i \in S}\sum_{j \in S} x_{ij} \leq |S|-1 \quad \forall S \subset \{ 1, \dots , n' \}
            \end{equation*}
            
            \begin{itemize}
                \item \scriptsize \textbf{Conjunto Potencia:} $2^{n'} - 2$ restricciones.
                \item \scriptsize \textbf{MTZ\footnote[frame]{Miller, Tucker y Zemlin (1960): Reformulación que introduce variables de orden $u_i$ para evitar subtours con complejidad $O(n'^2)$.} ($O(n'^2)$):} $u_i - u_j + n' x_{ij} \le n' - 1$.
            \end{itemize}

            \begin{block}{El Muro Computacional}
                \scriptsize Para $n' \approx 15,000$, incluso la versión polinomial genera $\approx 225$ millones de restricciones. El desbordamiento de RAM hace inviable el uso de solvers exactos.
            \end{block}
        \end{column}

        % Columna 2: Heurística de Dos Fases
        \begin{column}{0.45\textwidth}
            \textbf{Estrategia de Dos Fases}
            
            \vspace{0.2cm}
            \begin{enumerate}
                \item \small \textbf{Fase de Construcción:}
                \\ \footnotesize Algoritmo de \textbf{Inserción Arbitraria}. Selecciona un nodo al azar y lo inserta en el tour minimizando el incremento en la distancia total.
                
                \vspace{0.3cm}
                \item \small \textbf{Fase de Mejora (Refinamiento):}
                \\ \footnotesize Algoritmo \textbf{2-opt}. Aplica intercambios de aristas (\textit{2-edge exchange}) para optimizar el tour localmente.
            \end{enumerate}

            \begin{block}{Resultado en Opt-Art}
                \footnotesize La combinación genera tours fiables rápidamente. El 2-opt aporta a la \textbf{limpieza visual}, eliminando ruidosos cruces de líneas.
            \end{block}
            
            \vfill
            \scriptsize \textit{Conclusión: La heurística es la única vía de ejecución para $n' > 10^4$.}
        \end{column}
    \end{columns}
\end{frame}


% --- Slide 14: El Arte de Desanudar: El Efecto 2-opt ---
\begin{frame}{El Arte de Desanudar: El Efecto 2-opt}
    \begin{columns}[T]
        % Columna 1: Fundamento Geométrico
        \begin{column}{0.5\textwidth}
            El algoritmo \textbf{2-opt} intercambia dos aristas existentes por dos nuevas siempre que el costo total disminuya.
            
            \vspace{0.4cm}
            \begin{itemize}
                \item \textbf{Fundamento Euclidiano:} En el plano, la \textbf{Desigualdad Triangular} garantiza que ``descruzar'' dos aristas siempre resulta en una ruta más corta.
                \item \textbf{Impacto Visual:} Elimina el ruido visual de las intersecciones, ``peinando'' el trazo de forma iterativa.
                \item \textbf{Complejidad:} $O(n'^2)$ por iteración. Logra convergencia rápida hacia óptimos locales de alta fidelidad.
            \end{itemize}
        \end{column}

        % Columna 2: Diagrama TikZ y Sentencia
        \begin{column}{0.5\textwidth}
            \centering
            \begin{tikzpicture}[scale=0.85, transform shape]
                % Coordenadas

            \end{flushleft}

% --- Slide 22: Anexo A: Mapeo Biyectivo eta (Vectorización) ---
\begin{frame}{Anexo A: Mapeo Biyectivo $\eta$ (Vectorización)}
    \textbf{El Reto:} Traducir un tensor de costos $C_{f,i,j}$ a un vector $c^\top x$ compatible con solucionadores estándar.
    
    \vspace{0.4cm}
    \begin{algorithm}[H]
        \small
        \floatname{algorithm}{Algoritmo A1}
        \begin{algorithmic}[1]
            \Procedure{Mapeo$\eta$}{$f, i, j, m, n$}
                \State \Comment{$f$: imagen, $i,j$: posición, $m,n$: dimensiones malla}
                \State $t \gets (f - 1) \cdot (mn) + (i-1) \cdot n + j$
                \State \Return $t$
            \EndProcedure
        \end{algorithmic}
    \end{algorithm}
    
    \vspace{0.4cm}
    \textbf{Propiedades:}
    \begin{itemize}
        \item \small \textbf{Biyectividad:} Garantiza que cada decisión $(f, i, j)$ tenga una posición única en el vector.
        \item \small \textbf{Eficiencia:} Permite el direccionamiento directo en memoria, evitando búsquedas costosas durante la construcción de la matriz de restricciones.
    \end{itemize}
\end{frame}




% --- NUEVO Slide 23: Anexo B: Breviario de K-means ---
\begin{frame}{Anexo B: Breviario de $K$-means y Cuantización Vectorial}
    \textbf{El Objetivo:} Particionar un conjunto de $N$ observaciones en $K$ clusters, donde cada dato pertenece al centroide con la media más cercana.

    \begin{block}{Criterio de Optimización (Varianza Intra-cluster)}
        Dado un conjunto de datos $\mathbf{X} = \{\mathbf{x}_1, \dots, \mathbf{x}_N\}$, buscamos minimizar la suma de los cuadrados de las distancias a los centroides $\boldsymbol{\mu}_i$:
        \begin{equation*}
            \min_{\mathbf{S}} \sum_{i=1}^{K} \sum_{\mathbf{x}_j \in S_i} \left\| \mathbf{x}_j - \boldsymbol{\mu}_i \right\|^2
        \end{equation*}
    \end{block}

    \textbf{Fases del Algoritmo de Lloyd:}
    \begin{itemize}
        \item \small \textbf{Asignación:} Se asocia cada observación al cluster cuyo centroide es el más cercano (fronteras de Voronoi).
        \item \small \textbf{Actualización:} Se recalcula el centroide como la media aritmética de los puntos asignados:
        \begin{equation*}
            \boldsymbol{\mu}_i^{(t+1)} = \frac{1}{|S_i^{(t)}|} \sum_{\mathbf{x}_j \in S_i^{(t)}} \mathbf{x}_j
        \end{equation*}
    \end{itemize}
    \scriptsize \textit{Nota: En esta tesina se utiliza para extraer el color dominante $b_f$ en RGB y para el stippling en el plano $\mathbb{R}^2$.}
\end{frame}




% --- Slide 24: Anexo MatSplitter ---
\begin{frame}{Anexo C: Algoritmo de Partición Modular}
    \begin{algorithm}[H]
        \floatname{algorithm}{Algoritmo A1}
        \caption{Partición modular simétrica (\texttt{matsplitter})}
        \begin{algorithmic}[1]
            \Procedure{MatSplitter}{MatrizOriginal, m, n}
            \State $\mu \gets \text{nrow(MatrizOriginal)},\ \nu \gets \text{ncol(MatrizOriginal)}$
            \State $\mu' \gets \mu - (\mu \mod m)$ \Comment{Ajuste de filas}
            \State $\nu' \gets \nu - (\nu \mod n)$ \Comment{Ajuste de columnas}
            \State MatrizRecortada $\gets$ MatrizOriginal$[1:\mu', 1:\nu']$
            \State $k \gets \mu'/m,\ h \gets \nu'/n$ \Comment{Dimensiones del tile}
            \State ListaRectángulos $\gets$ Dividir(MatrizRecortada, k, h)
            \State \Return ListaRectángulos
            \EndProcedure
        \end{algorithmic}
    \end{algorithm}
    Garantiza bloques íntegros y evita sesgos de luminosidad en los bordes.
\end{frame}

% --- Slide 25: Anexo B: Extensión a Dominós de Bosch ---
\begin{frame}{Anexo D: Extensión a Dominós de Bosch}
    \textbf{El Reto:} Cubrir la región de la imagen mediante fichas de dominó ($2 \times 1$).
    
    \vspace{0.4cm}
    \begin{itemize}
        \item Se modela como un problema de \textbf{Emparejamiento Perfecto} en una gráfica de rejilla.
        \item Requiere la introducción de variables binarias de orientación: $x_{ij}^H$ (horizontal) y $x_{ij}^V$ (vertical).
    \end{itemize}

    \begin{block}{Restricción de Cobertura}
        Cada celda $(i,j)$ debe ser cubierta por exactamente una mitad de una ficha:
        \begin{equation*}
            x_{ij}^H + x_{i,j-1}^H + x_{ij}^V + x_{i-1,j}^V = 1
        \end{equation*}
    \end{block}

    \vfill

    \vspace{0.4cm}
\end{frame}

% --- Slide 27: Anexo F: Vector de Disponibilidades y Sensibilidad ---
\begin{frame}{Anexo F: Holgura ($hd$) y Variabilidad ($vd$) en $u$}
    \begin{columns}[T, onlytextwidth]
% Columna 1: Formulación del Vector u (CORREGIDA)
        \begin{column}{0.52\textwidth}
            \textbf{El Vector de Disponibilidades ($u$):} \\[0.15cm]
            \footnotesize Define el límite de uso para cada mosaico $f$. Su construcción es una propuesta original de este trabajo.
            
            \vspace{0.5cm}
            
            \begin{centering}
                \textbf{\small Muestreo de la dotación $u_f$:} \\
                \vspace{0.2cm}
                \footnotesize $u_{f} \in \left\{ \left\lceil \frac{mn(1+hd)}{cf}(1-vd) \right\rceil , \dots , \left\lceil \frac{mn(1+hd)}{cf}(1+vd) \right\rceil \right\}$ \\
            \end{centering}
            
            \vspace{0.5cm}
            
            \begin{itemize}
                \item \footnotesize \textbf{Holgura ($hd \ge 0$):} Exceso de inventario respecto al área $m \cdot n$. Garantiza que la región factible sea no vacía: $\mathcal{F} \neq \emptyset$.
                \item \footnotesize \textbf{Variabilidad ($vd \in [0,1]$):} Grado de dispersión en la distribución. Si $vd=0$, la disponibilidad es estrictamente homogénea.
            \end{itemize}
            \vfill
        \end{column}

        
        % Columna 2: Análisis de Sensibilidad (Caso Twitter)
        \begin{column}{0.45\textwidth}
            \textbf{Análisis de Sensibilidad:} \\
            \footnotesize En la imagen de \textbf{Twitter}, la escasez de mosaicos blancos ($u_f$ reducido) fuerza el uso de grises claros para mantener la factibilidad global.
            
            \vspace{0.3cm}
            
            \begin{block}{Caso Extremo: Disponibilidad Total ($hd = mn$)}
                \footnotesize
                Al liberar la restricción de escasez (cada mosaico puede cubrir toda la imagen), el error residual $z^*$ disminuye drásticamente:
                \vspace{0.2cm}
                \begin{itemize}
                    \item \textbf{Sin restricción:} $z^* \approx 0.15\%$ del supremo.
                    \item \textbf{Con restricción:} $z^* \approx 3.00\%$ del supremo.
                \end{itemize}
                \vspace{0.1cm}
                \centering \scriptsize \textit{Mejora de fidelidad de hasta 20 veces.}
            \end{block}
            
            \vfill
            
            \centering
            \alert{\footnotesize \textbf{``La escasez impone la estética: El modelo elige el 'menor mal' ante recursos limitados.''}}
            \vspace{0.2cm}
        \end{column}
    \end{columns}
\end{frame}
